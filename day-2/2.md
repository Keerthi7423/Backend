## рЯІ© app.js (MAIN LOGIC)

const { readContacts, writeContacts } = require("./utils/fileHandler");

const command = process.argv[2];
const name = process.argv[3];
const phone = process.argv[4];

let contacts = readContacts();

// ADD CONTACT
if (command === "add") {
if (!name || !phone) {
console.log("Invalid input вЭМ");
return;
}

const exists = contacts.find(c => c.name === name);
if (exists) {
console.log("Contact already exists вЭМ");
return;
}

contacts.push({ name, phone });
writeContacts(contacts);
console.log("Contact added вЬЕ");
}

// LIST CONTACTS
else if (command === "list") {
console.log(contacts);
}

// DELETE CONTACT
else if (command === "delete") {
const updated = contacts.filter(c => c.name !== name);
writeContacts(updated);
console.log("Contact deleted вЭМ");
}

   вЖТ It is an array that contains command-line arguments passed to a Node.js program.

4. Difference between JSON.parse() and JSON.stringify()?
   вЖТ JSON.parse() converts JSON вЖТ JS object

   вЖТ Using databases (SQL, MongoDB) or files (JSON for small apps)

7. How to read a file in Node.js?
   вЖТ fs.readFileSync()

8. How to write a file in Node.js?
   вЖТ fs.writeFileSync()
